GDScript - Interactive Chest System

GDScript implementation of an interactive chest that contains collectible items:

extends Node2D
class_name Chest

@export var items: Array = []
@export var isOpen: bool = false
@onready var sprite = $Sprite2D
@onready var animation_player = $AnimationPlayer

signal chest_opened
signal item_collected(item)

func _ready():
    sprite.frame = 0 if not isOpen else 1
    add_to_group("interactive_objects")

func _input_event(viewport, event, shape_idx):
    if event is InputEventMouseButton and event.pressed:
        if not isOpen:
            open_chest()

func open_chest():
    isOpen = true
    animation_player.play("open")
    chest_opened.emit()
    
    # Spawn collectible items
    for item in items:
        spawn_item(item)
    
    # Disable further interaction
    input_pickable = false

func spawn_item(item_name: String):
    var item = preload("res://scenes/collectible.tscn").instantiate()
    item.item_name = item_name
    item.global_position = global_position + Vector2(randf_range(-30, 30), -50)
    get_parent().add_child(item)
    item_collected.emit(item_name)

GDScript - Tile Map Manager

Tile map manager for handling level loading and rendering:

extends TileMap
class_name LevelManager

@export var level_name: String = ""
@export var enable_collision: bool = true

var tile_size = 16
var layer_indices = {
    "ground": 0,
    "obstacles": 1,
    "decorations": 2
}

func _ready():
    load_level(level_name)
    if enable_collision:
        setup_collision_physics()

func load_level(level: String):
    # Load tile map data based on level name
    print("Loading level: ", level)
    # Tiles are set in the Godot editor
    update_visibility()

func setup_collision_physics():
    # Create physics bodies for solid tiles
    for layer_index in layer_indices.values():
        for cell in get_used_cells(layer_index):
            var tile_data = get_cell_tile_data(layer_index, cell)
            if tile_data and tile_data.get_custom_data("solid"):
                add_collision_body(cell, layer_index)

func add_collision_body(cell: Vector2i, layer: int):
    var world_pos = map_to_local(cell)
    var area = Area2D.new()
    area.global_position = world_pos
    area.name = "Collision_" + str(cell)
    add_child(area)

func update_visibility():
    show()
    print("Level '%s' loaded with %d tiles" % [level_name, get_used_cells(0).size()])

JavaScript - Game Event Manager

JavaScript implementation for managing game events and callbacks:

class GameEventManager {
    constructor() {
        this.listeners = new Map();
        this.eventHistory = [];
    }

    // Subscribe to events
    on(eventName, callback) {
        if (!this.listeners.has(eventName)) {
            this.listeners.set(eventName, []);
        }
        this.listeners.get(eventName).push(callback);
    }

    // Emit events
    emit(eventName, data = null) {
        if (this.listeners.has(eventName)) {
            const callbacks = this.listeners.get(eventName);
            callbacks.forEach(callback => callback(data));
        }
        this.eventHistory.push({
            name: eventName,
            data: data,
            timestamp: Date.now()
        });
    }

    // Remove event listener
    off(eventName, callback) {
        if (this.listeners.has(eventName)) {
            const callbacks = this.listeners.get(eventName);
            const index = callbacks.indexOf(callback);
            if (index > -1) {
                callbacks.splice(index, 1);
            }
        }
    }

    // Get event history
    getHistory(eventName = null) {
        return eventName 
            ? this.eventHistory.filter(e => e.name === eventName)
            : this.eventHistory;
    }
}

// Usage example
const gameEvents = new GameEventManager();
gameEvents.on("chest_opened", (item) => {
    console.log("Chest opened with item:", item);
});
gameEvents.emit("chest_opened", {itemName: "Gold Key"});
