Learning Whiz

An interactive educational app for elementary students with quizzes and adventure stories

Overview & Summary

Learning Whiz is an educational JavaScript application designed to engage elementary school students in learning through interactive quizzes and choose-your-own-adventure story modes. Developed as a Code.org App Lab project with real classroom testing and feedback integration, Learning Whiz demonstrates the ability to design educational software that is both engaging and effective. The project was developed independently with mentor guidance, incorporating actual student feedback from classroom testing to refine the learning experience.

JavaScript - Quiz Management System

// Quiz question data structure
const quizQuestions = [
    {
        question: "What is 5 + 3?",
        options: ["6", "8", "9", "7"],
        correct: 1
    },
    {
        question: "Which animal is a mammal?",
        options: ["Fish", "Bird", "Dog", "Snake"],
        correct: 2
    }
];

let currentQuestion = 0;
let score = 0;

function displayQuestion() {
    const q = quizQuestions[currentQuestion];
    setText("questionText", q.question);
    for (let i = 0; i < q.options.length; i++) {
        setText("option" + i, q.options[i]);
    }
}

function checkAnswer(selected) {
    if (selected === quizQuestions[currentQuestion].correct) {
        score++;
        showMessage("Correct! Great job!");
    } else {
        showMessage("Try again!");
    }
    currentQuestion++;
    if (currentQuestion < quizQuestions.length) {
        displayQuestion();
    } else {
        showFinalScore();
    }
}

JavaScript - Choose-Your-Own-Adventure Logic

// Story state management
const storyPaths = {
    start: {
        text: "You wake up in a magical forest. What do you do?",
        choices: [
            { text: "Follow the blue light", next: "blueLight" },
            { text: "Explore the forest", next: "explore" }
        ]
    },
    blueLight: {
        text: "You follow the mysterious blue light and find a wise wizard!",
        choices: [
            { text: "Ask for help", next: "wizardHelp" },
            { text: "Be cautious", next: "cautious" }
        ]
    },
    explore: {
        text: "You discover a hidden village of friendly creatures.",
        choices: [
            { text: "Make friends", next: "makeFriends" },
            { text: "Continue exploring", next: "continueExploring" }
        ]
    }
};

let currentScene = "start";

function displayScene() {
    const scene = storyPaths[currentScene];
    setText("storyText", scene.text);
    clearScreen("choiceButtons");
    
    for (let i = 0; i < scene.choices.length; i++) {
        const button = createButton(scene.choices[i].text);
        setOnClick(button, function() {
            currentScene = scene.choices[i].next;
            displayScene();
        });
        append("choiceButtons", button);
    }
}

displayScene();

JavaScript - Student Progress Tracking

// Student performance tracking
let studentStats = {
    name: "",
    quizzesTaken: 0,
    totalScore: 0,
    averageScore: 0,
    storiesCompleted: 0
};

function updateStats(quizScore) {
    studentStats.quizzesTaken++;
    studentStats.totalScore += quizScore;
    studentStats.averageScore = 
        studentStats.totalScore / studentStats.quizzesTaken;
}

function completeStory() {
    studentStats.storiesCompleted++;
    const feedbackMessage = 
        "Congratulations! You completed the adventure!";
    showMessage(feedbackMessage);
    displayStats();
}

function displayStats() {
    setText("statsDisplay", 
        "Quizzes: " + studentStats.quizzesTaken + 
        " | Avg Score: " + 
        Math.round(studentStats.averageScore) + "%"
    );
}
